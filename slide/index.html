Зависимости в KnockOut
function AppViewModel( ){
    this.firstName = ko.observable( 'Bert' )
    this.lastName = ko.observable( 'Bertington' )
    this.fullName = ko.computed( function( ){
        return this.firstName() + ' ' + this.lastName()
    }, this )
}
Зависимости в AngularJS
function AppViewModel( $scope ){
    $scope.firstName = 'Bert'
    $scope.lastName = 'Bertington'
    $scope.fullName = ''
    $scope.$watch( function( ){
        return $scope.firstName + ' ' + $scope.lastName
    }, function( next ){
        this.fullName = next
    })
}

Привязки в $jin.atom
$jin.klass({ 'MyFriends.list': [ '$jin.view' ] })
$jin.atom.prop({ 'MyFriends.list..names': { value: [ 'Bart', 'John' ] } })
$jin.atom.prop({ 'MyFriends.list..friendViews': { pull: function( ){
  return this.names().map( function( name, index ){
    return this.make( 'friend=' + index, MyFriends.item ).name( name )
  }.bind( this ) )
} } })

$jin.klass({ 'MyFriends.item': [ '$jin.view' ] })
$jin.atom.prop({ 'MyFriends.item..name': { merge: String } })
